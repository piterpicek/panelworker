Basic

<!-- map -->
<div id="map1" class="map"></div>
<!-- /map -->

<script>
/*
 * MAP 1
 */
jQuery(function() {
  $('#map1').vectorMap({
    map: 'world_mill_en',
    series: {
      regions: [{
        values: gdpData,
        scale: ['#C8EEFF', '#0071A4'],
        normalizeFunction: 'polynomial'
      }]
    },
    onRegionTipShow: function(e, el, code){
      el.html(el.html()+' (GDP - '+gdpData[code]+')');
    }
  });
})
</script>
